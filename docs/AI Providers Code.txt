            model.addEventListener('click', function() {
                const card = this.closest('.provider-card');
                const bestFor = card.querySelector('.best-for');
                bestFor.classList.toggle('visible');
            });
        });
        
        // Toggle all model details at once
        let allVisible = false;
        function toggleAll() {
            const allBestFor = document.querySelectorAll('.best-for');
            const btn = document.querySelector('.toggle-all-btn');
            
            if (allVisible) {
                allBestFor.forEach(bf => bf.classList.remove('visible'));
                btn.textContent = 'Show All Model Details';
                allVisible = false;
            } else {
                allBestFor.forEach(bf => bf.classList.add('visible'));
                btn.textContent = 'Hide All Model Details';
                allVisible = true;
            }
        }
    </script>
